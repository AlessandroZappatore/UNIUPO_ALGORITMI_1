138size_t upo_quick_sort_partition(void *base, size_t lo, size_t hi, size_t size, upo_sort_comparator_t cmp)
139{
140 size_t p = lo, i = lo, j = hi + 1;
141 unsigned char *ptr = base;
142
143 while (1)
144 {
145 do
146 {
147 i += 1;
148 } while (i <= hi && cmp(ptr + i * size, ptr + p * size) < 0);
149
150 do
151 {
152 j -= 1;
153 } while (j >= lo && cmp(ptr + j * size, ptr + p * size) > 0);
154
155 if (i >= j)
156 {
157 break;
158 }
159
160 upo_swap(ptr + i * size, ptr + j * size, size);
161 }
162
163 upo_swap(ptr + p * size, ptr + j * size, size);
164 return j;
165}
